[docs]def popen_py(cmdlist, usepyshell=False):
    """Popen python script."""
    if usepyshell:
        p = pyshellexe()
    else:
        p = pyexe()
    fullcmdlist = [p]
    fullcmdlist.extend(cmdlist)
    return popen(fullcmdlist)

[docs]def call(cmdlist):
    """Call cmdlist."""
    try:
        ret = subprocess.check_output(cmdlist, stderr=subprocess.STDOUT)
        output = ret
        retcode = 0
    except subprocess.CalledProcessError as err:
        output = err.output
        retcode = err.returncode
        print('ERROR')
    print(retcode)
    print(output)
    return retcode
